// This function creates and returns a new person object with the given properties
function newPerson(firstName, lastName, age, eyeColour) {
    return {
        firstName: firstName,
        lastName: lastName,
        age: age,
        eyeColour: eyeColour
    };
}

// We can now create a person object using the template, to do this we use the new keyword
let person1 = newPerson("John", "Doe", 50, "blue"); // Create a person object
console.log(typeof person1);
console.log("ok");

// We can also create an array of people
let people = [newPerson("John", "Doe", 50, "blue"), newPerson("Jane", "Doe", 40, "green"), newPerson("Jack", "Doe", 30, "brown")]; // Create an array of people

// Like any array we can iterate through the array of objects using a for loop
for (let i = 0; i < people.length; i++) { // Loop through the array
    let person = people[i]; // Get the current person
    console.log("Hello " + person.firstName); // Print the first name of the current person
}

// We can also create a string of all the names
let allNames = ""; // Create a variable to hold all the names
for (let i = 0; i < people.length; i++) { // Loop through the array
    let person = people[i]; // Get the current person
    allNames += person.firstName + " "; // Add the first name of the current person to the allNames variable
}
console.log("Hello " + allNames); // Print all the names at once
